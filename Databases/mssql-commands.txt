RESTORE DATABASE DBbienestar FROM DISK = N'/var/opt/mssql/data/DBbienestar.bak'
WITH MOVE 'DBbienestar' TO '/var/opt/mssql/data/DBbienestar.mdf',

// GENERAR BACKUP DE BASE DE DATOS
BACKUP DATABASE DBbienestar
